\chapter{Metatheory: Normalization, Decidability, and Conservativity}
\label{ch:c4-metatheory}

This chapter establishes the metatheoretic properties of $\mathrm{C}^4$:
strong normalization, decidability of type checking, and conservativity
over both CIC and cubical type theory.  We conclude with a systematic
comparison to other verification systems and a conjecture about the
universality of $\mathrm{C}^4$.


% ─────────────────────────────────────────────────────────
\section{Strong Normalization}
\label{sec:normalization}

\begin{theorem}[Strong Normalization of $\mathrm{C}^4$]
\label{thm:strong-normalization}
Every well-typed term in $\mathrm{C}^4$ has a normal form under the
computation rules of Chapter~\ref{ch:c4-calculus}.
\end{theorem}

\begin{proof}
We decompose $\mathrm{C}^4$ into four fragments and show that each
normalizes, then show that the combined reduction relation is terminating.

\textbf{Fragment 1: CIC.}
The standard $\beta\delta\iota$-reduction of the Calculus of Inductive
Constructions is strongly normalizing by the theorem of
Coquand--Paulin~\cite{paulin1993}.  This covers $\Pi$-types,
$\Sigma$-types, inductive type recursors, and universe cumulativity.

\textbf{Fragment 2: Cubical.}
The cubical computation rules---$\Path$-$\beta$, $\Path$-$\eta$,
$\transp$ on type formers, and $\hfilling$ computation---are strongly
normalizing by the result of Huber~\cite{huber2018}.  The key insight is
that $\transp$ on a constant type family is the identity, and transport
on each type former reduces to transport on its components, each of which
is at a smaller type.

\textbf{Fragment 3: Descent and restriction.}
The descent and restriction computation rules are:
\begin{enumerate}
\item $\restr{(\lambda x.\, t)}{U} \rightsquigarrow
  \lambda x.\, \restr{t}{U}$ \quad (restriction distributes)
\item $\restr{(t, u)}{U} \rightsquigarrow
  (\restr{t}{U}, \restr{u}{U})$ \quad (restriction distributes)
\item $\restr{(\langle i \rangle\, t)}{U} \rightsquigarrow
  \langle i \rangle\, \restr{t}{U}$ \quad (restriction distributes)
\item $\restr{\mathsf{desc}(\{t_\alpha\}, \{p_{\alpha\beta}\})}{U_\alpha}
  \rightsquigarrow t_\alpha$ \quad (descent-$\beta$)
\end{enumerate}

These rules are terminating because:
\begin{itemize}
\item Rules (1)--(3) push restriction inward, decreasing the term size
  above the restriction operator (the subterm $t$ is exposed for further
  reduction; restriction cannot create new redexes above itself).
\item Rule (4) eliminates the $\mathsf{desc}$ constructor, replacing it
  with a local section $t_\alpha$ which is smaller.
\end{itemize}

We formalize this with the lexicographic measure:
$\mu(t) = (\text{number of } \mathsf{desc}\text{ nodes},\;
\text{depth of outermost } \restr{}{}\text{ not under }\mathsf{desc},\;
|t|)$.
Each rule strictly decreases $\mu$ in the lexicographic order, so
the reduction is well-founded.

\textbf{Fragment 4: OTerm axioms.}
Axiom computation $\den{\ell\sigma} \rightsquigarrow \den{r\sigma}$
applies when a CCTT axiom $\ell \equiv r$ matches.  These are
terminating because:
\begin{enumerate}
\item The axiom set is finite (24 dichotomy axioms + 48 pattern axioms).
\item Each axiom application is oriented: the left-hand side $\ell$ has
  a designated ``redex form'' and the right-hand side $r$ is in a more
  ``canonical'' form.  The orientation is compatible with a term ordering
  (e.g., Knuth--Bendix ordering on the OTerm constructors).
\item No axiom increases the size of the term beyond a constant factor.
\end{enumerate}

\textbf{Confluence of the combined system.}
We claim that the four fragment reductions commute (no critical pairs
between fragments lead to divergence):
\begin{itemize}
\item CIC and cubical reductions commute by the result of
  Cohen et al.~\cite{cohen2018}.
\item Restriction distributes uniformly over all constructors (CIC,
  cubical, and OTerm), so restriction rules commute with all others.
\item OTerm axiom rules operate only on $\den{\cdot}$ subterms and
  do not interact with CIC/cubical constructors.
\end{itemize}

Since the fragments commute and each is terminating, the combined
reduction is terminating (by the modular termination theorem).
\end{proof}

\section{Decidability of Type Checking}
\label{sec:c4-decidability}

\begin{theorem}[Decidability of Type Checking]
\label{thm:decidability}
Given a context $\Gamma$ and terms $t$, $A$, it is decidable whether
$\Gamma \vdash t : A$ in $\mathrm{C}^4$.
\end{theorem}

\begin{proof}
Type checking in $\mathrm{C}^4$ reduces to the following decidable
sub-problems:

\textbf{(a) Definitional equality is decidable.}
To check $\Gamma \vdash t \equiv u : A$, reduce both $t$ and $u$ to
normal form (which exists by Theorem~\ref{thm:strong-normalization})
and compare syntactically.  Strong normalization guarantees that
normalization terminates; confluence guarantees that the normal forms are
unique.  Hence definitional equality is decidable.

\textbf{(b) Type inference is decidable.}
$\mathrm{C}^4$ uses bidirectional type checking (as in Lean~4 and
cubical Agda).  The inference rules for $\Pi$-intro, $\Pi$-elim,
$\Sigma$-intro/elim, $\Path$-intro/elim are syntax-directed: each
term former has exactly one applicable rule.  The fiber restriction rule
$\restr{t}{U}$ requires inferring the type of $t$ (a recursive call)
and then restricting.

\textbf{(c) Fiber restriction checking is decidable.}
Given $U \in \mathrm{Ob}(\site)$, checking $\Gamma \vdash_U t :
\restr{A}{U}$ reduces to checking $\Gamma \vdash t : A$ in the
restricted context.  Since the site $\site_\mathrm{CCTT}$ is finite
(Definition~\ref{def:cctt-site}), there are finitely many fibers to
consider.

\textbf{(d) Descent checking is decidable.}
To check $\mathsf{desc}(\{t_\alpha\}, \{p_{\alpha\beta}\}) : A$,
we must verify:
\begin{enumerate}
\item Each $t_\alpha : \restr{A}{U_\alpha}$ --- finitely many checks.
\item Each $p_{\alpha\beta}$ witnesses compatibility --- finitely many
  equality checks.
\item The cocycle condition on triple overlaps --- finitely many
  (at most $|\covers|^3$) equality checks.
\end{enumerate}
Each is a decidable equality check (by part~(a)), and there are finitely
many of them.

\textbf{(e) $\Hone$ computation is decidable.}
Over the finite site $\site_\mathrm{CCTT}$, the \v{C}ech complex has
finitely many terms at each level.  Computing $\Hone$ reduces to
solving a finite system of linear equations (in the abelian case) or
checking a finite list of cocycle conditions (in the non-abelian case).
Both are decidable.

Combining (a)--(e), all sub-tasks of type checking are decidable.
\end{proof}

\section{Conservativity over CIC}
\label{sec:conservativity-cic}

We now give the full proof of the conservative extension theorem
sketched in Chapter~\ref{ch:c4-calculus}.

\begin{definition}[Erasure Map $\varepsilon$]
\label{def:erasure-map}
The \textbf{erasure map} $\varepsilon : \mathrm{C}^4 \to \mathrm{CIC}$
is defined by structural recursion:
\begin{align}
  \varepsilon(\mathcal{U}_i^{\site}) &\defeq \mathcal{U}_i
    \tag{erase site annotation} \\
  \varepsilon(\Pi(x:A).\,B) &\defeq
    \Pi(x:\varepsilon(A)).\,\varepsilon(B) \\
  \varepsilon(\Sigma(x:A).\,B) &\defeq
    \Sigma(x:\varepsilon(A)).\,\varepsilon(B) \\
  \varepsilon(\lambda(x:A).\,t) &\defeq
    \lambda(x:\varepsilon(A)).\,\varepsilon(t) \\
  \varepsilon(t\; u) &\defeq \varepsilon(t)\; \varepsilon(u) \\
  \varepsilon(\Path_A\, a\, b) &\defeq
    \mathsf{Id}_{\varepsilon(A)}\, \varepsilon(a)\, \varepsilon(b)
    \tag{paths $\to$ identity} \\
  \varepsilon(\langle i \rangle\, t) &\defeq
    \mathsf{refl}\; [\text{when $t$ is constant in $i$}]
    \tag{path abstraction} \\
  \varepsilon(\transp^i\, A\, t) &\defeq
    \mathsf{J}(\ldots, \varepsilon(t))
    \tag{transport $\to$ J} \\
  \varepsilon(\hfilling^i\, A\, [\vec{s}]\, t_0) &\defeq
    \varepsilon(t_0)
    \tag{hcomp erases to base} \\
  \varepsilon(\restr{t}{U}) &\defeq \varepsilon(t)
    \tag{erase restriction} \\
  \varepsilon(\mathsf{desc}(\{t_\alpha\}, \{p_{\alpha\beta}\}))
    &\defeq \varepsilon(t_{\alpha_0})
    \tag{descent $\to$ pick first section} \\
  \varepsilon(\den{P}) &\defeq c_P
    \tag{OTerm $\to$ fresh constant} \\
  \varepsilon(\mathsf{import}_\mathrm{Lean}(n)) &\defeq c_n
    \tag{axiom $\to$ fresh constant}
\end{align}

For contexts: $\varepsilon(\Gamma, x :_U A) \defeq
\varepsilon(\Gamma), x : \varepsilon(A)$ (erase fiber annotation).
\end{definition}

\begin{theorem}[Conservativity --- Full Proof]
\label{thm:conservativity-full}
$\mathrm{C}^4$ is a conservative extension of CIC:
\begin{enumerate}
\item \textbf{(Forward embedding.)} If $\Gamma \vdash_\mathrm{CIC}
  t : A$, then $\Gamma \vdash_{\mathrm{C}^4} t : A$ (viewing CIC
  terms as $\mathrm{C}^4$ terms with trivial site).

\item \textbf{(Reverse preservation.)} If $\Gamma$, $t$, $A$ are all
  in the CIC fragment (no restriction, no descent, no OTerms, no Lean
  imports, no path types) and $\Gamma \vdash_{\mathrm{C}^4} t : A$,
  then $\varepsilon(\Gamma) \vdash_\mathrm{CIC} \varepsilon(t) :
  \varepsilon(A)$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{Part (1):} CIC rules are a subset of $\mathrm{C}^4$ rules.
Any CIC derivation is a valid $\mathrm{C}^4$ derivation with the
trivial site $\site_{\mathrm{triv}} = \{\ast\}$ (one object, one
morphism).  Under $\site_\mathrm{triv}$:
\begin{itemize}
\item $\mathcal{U}_i^{\site_\mathrm{triv}} \equiv \mathcal{U}_i$
  (only one fiber, which is the whole universe).
\item $\restr{t}{\ast} \equiv t$ (restriction is identity).
\item $\mathsf{desc}(\{t_\ast\}, \{\}) \equiv t_\ast$ (descent with
  one element is trivial).
\end{itemize}

\textbf{Part (2):}  We show that $\varepsilon$ preserves typing by
induction on the $\mathrm{C}^4$ derivation $\mathcal{D}$ of
$\Gamma \vdash_{\mathrm{C}^4} t : A$.

Since $\Gamma$, $t$, $A$ are in the CIC fragment, the derivation
$\mathcal{D}$ can only use:
\begin{itemize}
\item CIC rules ($\Pi$F, $\Pi$I, $\Pi$E, $\Sigma$F, Ind, etc.).
  These map directly to CIC rules under $\varepsilon$ (which acts as
  identity on CIC terms).

\item The universe rule (Univ) with site annotation.  But since $t$
  and $A$ are in the CIC fragment, the site must be trivial,
  and $\varepsilon(\mathcal{U}_i^{\site_\mathrm{triv}}) = \mathcal{U}_i$.

\item Conversion rule: $\Gamma \vdash t : A$ from $\Gamma \vdash t : A'$
  and $\Gamma \vdash A' \equiv A$.  The additional definitional equalities
  in $\mathrm{C}^4$ (restriction distribution, descent-$\beta$, axiom-$\beta$)
  cannot produce new equalities between CIC terms, because:
  \begin{itemize}
  \item Restriction rules only apply to terms containing $\restr{}{}$,
    which CIC terms do not contain.
  \item Descent-$\beta$ only applies to $\mathsf{desc}$ terms, absent
    from the CIC fragment.
  \item Axiom-$\beta$ only applies to $\den{\cdot}$ terms, absent from
    CIC.
  \end{itemize}
  Therefore, $A' \equiv_{\mathrm{C}^4} A$ for CIC terms implies
  $A' \equiv_\mathrm{CIC} A$.
\end{itemize}

In every case, the $\mathrm{C}^4$ derivation step maps to a valid CIC
derivation step under $\varepsilon$.  By induction, the full derivation
is preserved.
\end{proof}

\begin{corollary}
Any Lean theorem, imported via $\Phi$, does not create new derivable
judgments in the CIC fragment.
\end{corollary}

\begin{proof}
Lean imports become fresh constants $c_n$ under $\varepsilon$.  Fresh
constants do not inhabit any pre-existing CIC type (they have their own
types $\varepsilon(\Phi(P))$ which are new axiom types).  Therefore,
no existing CIC judgment gains a new derivation.
\end{proof}

\section{Conservativity over Cubical Type Theory}
\label{sec:conservativity-cchm}

\begin{theorem}[Conservativity over CCHM]
\label{thm:conservativity-cchm}
$\mathrm{C}^4$ restricted to the trivial site $\site_\mathrm{triv}$
and without OTerm/Lean-import primitives is exactly CCHM cubical type
theory.
\end{theorem}

\begin{proof}
Under $\site_\mathrm{triv}$:
\begin{enumerate}
\item All site-decorated universes reduce to plain universes.
\item The restriction rule (Res) becomes trivial ($\restr{t}{\ast} = t$).
\item The descent rule with a single cover element reduces to the
  identity (any section over the single object is already global).
\item The pullback rule (Pull) has only the identity morphism, which
  acts as identity.
\end{enumerate}

What remains is exactly the CCHM system: universes, $\Pi$, $\Sigma$,
inductive types, interval $\interval$ with connections, $\Path$ types,
$\transp$, $\hfilling$, and $\mathsf{Glue}$.  The typing and computation
rules match Cohen--Coquand--Huber--M\"ortberg~\cite{cohen2018} exactly.
\end{proof}

\section{Confluence}
\label{sec:confluence}

\begin{theorem}[Confluence of $\mathrm{C}^4$ Reduction]
\label{thm:c4-confluence}
The reduction relation $\rightsquigarrow$ of $\mathrm{C}^4$ is confluent:
if $t \rightsquigarrow^* u_1$ and $t \rightsquigarrow^* u_2$, then there
exists $v$ with $u_1 \rightsquigarrow^* v$ and $u_2 \rightsquigarrow^* v$.
\end{theorem}

\begin{proof}[Proof sketch]
We verify confluence by checking that all critical pairs are joinable.

\textbf{CIC critical pairs:} joinable by the standard Church--Rosser
theorem for CIC.

\textbf{Cubical critical pairs:} joinable by the confluence result for
CCHM~\cite{huber2018}.

\textbf{Restriction critical pairs:}
\begin{itemize}
\item $\restr{((\lambda x.\, t)\; u)}{U}$: can reduce the $\beta$-redex
  first (giving $\restr{t[x:=u]}{U}$) or distribute restriction first
  (giving $(\lambda x.\, \restr{t}{U})\; \restr{u}{U}$, then
  $\beta$-reducing to $\restr{t}{U}[x := \restr{u}{U}]$).  These are
  equal because $\restr{t[x:=u]}{U} = \restr{t}{U}[x := \restr{u}{U}]$
  (restriction commutes with substitution, provable by structural
  induction on $t$).
\item $\restr{(\mathsf{desc}(\ldots))}{U_\alpha}$ composed with
  restriction distribution: by descent-$\beta$, this gives $t_\alpha$,
  which is the same regardless of which rule fires first.
\end{itemize}

\textbf{Axiom critical pairs:} The OTerm axiom set is oriented and
non-overlapping (no two axiom left-hand sides overlap at the same subterm
position, a property verified by the compilation step that generates
the rewrite system).  Therefore, axiom reduction has no critical pairs.

\textbf{Cross-fragment pairs:} The four fragments operate on disjoint
term constructors ($\lambda/\Pi/\Sigma$ for CIC; $\langle i
\rangle/\transp/\hfilling$ for cubical; $\restr{}{}/\mathsf{desc}$ for
sheaf; $\den{}$ for OTerm).  The only interactions are through
restriction distribution (handled above).

All critical pairs are joinable, so $\rightsquigarrow$ is locally
confluent.  Combined with strong normalization
(Theorem~\ref{thm:strong-normalization}), Newman's lemma gives
confluence.
\end{proof}

\section{The Initiality Conjecture}
\label{sec:initiality}

\begin{conjecture}[Initiality of $\mathrm{C}^4$]
\label{conj:initiality}
$\mathrm{C}^4$ is the initial object in the category $\mathbf{CubCoh}$
of type theories that have both cubical path structure (interval, Kan
operations, univalence) and sheaf descent (site-decorated universes,
restriction, descent rule, cohomological obstruction).

Concretely: for any type theory $T$ in $\mathbf{CubCoh}$, there exists
a unique interpretation functor $\mathrm{C}^4 \to T$ that preserves
the cubical and cohomological structure.
\end{conjecture}

\begin{remark}
The conjecture is motivated by the analogy with CIC being initial among
type theories with inductive types, and CCHM being initial among cubical
type theories.  $\mathrm{C}^4$ adds the minimal cohomological structure
(site-decorated universes + descent) to CCHM, suggesting it is the
``free'' extension.

A proof would proceed by showing that the rules of $\mathrm{C}^4$ are
exactly the equations that must hold in any model of $\mathbf{CubCoh}$,
and that no additional equations are imposed.  We leave this as an open
problem.
\end{remark}
